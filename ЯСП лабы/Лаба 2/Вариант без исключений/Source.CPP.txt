/*
Канатников Максим 9 группа
8. Создать класс Time для работы со временем в формате «час:минута:секунда». Класс
должен включать в себя не менее четырех способов инициализации: числами,
строкой (например, «23:59:59»), секундами, временем. Обязательными операциями
являются: вычисление разности между двумя моментами времени в секундах,
сложение времени и заданного количества секунд, вычитание из времени заданного
количества секунд, сравнение моментов времени, перевод в секунды, перевод
в минуты (с округлением до целой минуты).
*/

#include "UserInteraction.h"
#include "Time.h"
#include <Windows.h>


// действия над моментом времени
void action(Time& time);

void console_output(Time& time);

void file_output(Time& time, std::ofstream& file);

// выбор способа инициализации и инициализация
int initialization(Time& time);

void read_file_nums(std::ifstream& file, uShort& hour, uShort& minute, uShort& second);
void read_file_str(std::ifstream& file, std::string& str);
void read_file_sec(std::ifstream& file, uShort& sec);

void read_console_nums(uShort& hour, uShort& minute, uShort& second);
void read_console_str(std::string& str);
void read_console_sec(uShort& sec);


int main()
{
	SetConsoleOutputCP(1251);
	int choice = 0;	
	do
	{
		Time time;
		choice = initialization(time);
		if (choice)
			action(time);
	} while (choice != 0);
	return 0;
}

int initialization(Time& time)
{
	int choice = inizialization_menu();
 	switch (choice)
	{
	case 1: // числами
	{
		uShort hour, minute, second;
		switch(input_menu())
		{
		case 1:
			read_console_nums(hour, minute, second);
			time = Time(hour, minute, second);
			break;
		case 2:
		{
			std::string fname = choose_in_file_name();
			std::ifstream file(fname);
			read_file_nums(file, hour, minute, second);
			file.close();
			time = Time(hour, minute, second);
			console_output(time);
			break;
		}
		}	
		break;
	}
	case 2: // строкой
	{
		std::string str;
		switch (input_menu())
		{
		case 1:
			read_console_str(str);
			time = Time(str);
			break;
		case 2:
		{
			std::string fname = choose_in_file_name();
			std::ifstream file(fname);
			read_file_str(file, str);
			file.close();
			time = Time(str);
			console_output(time);
			break;
		}
		}
		break;
	}
	case 3: // секундами
	{
		uShort sec;
		switch (input_menu())
		{
		case 1:
			read_console_sec(sec);
			time = Time(sec);
			break;
		case 2:
		{
			std::string fname = choose_in_file_name();
			std::ifstream file(fname);
			read_file_sec(file, sec);
			file.close();
			time = Time(sec);
			console_output(time);
			break;
		}
		}
		break;
	}
	case 4: // временем
	{
		uShort hour, minute, second;
		switch (input_menu())
		{
		case 1:
		{
			read_console_nums(hour, minute, second);
			tm t{ second, minute, hour };
			time = Time(t);
			break;
		}
		case 2:
		{
			std::string fname = choose_in_file_name();
			std::ifstream file(fname);
			read_file_nums(file, hour, minute, second);
			file.close();
			tm t{ second, minute, hour };
			time = Time(t);
			console_output(time);
			break;
		}
		}
		break;
	}
	}
	return choice;
}

void read_file_nums(std::ifstream& file, uShort& hour, uShort& minute, uShort& second)
{
	while (!(file >> hour) || !(file >> minute) || !(file >> second) || hour >= 24 || minute >= 60 || second >= 60)
	{
		file.close();
		std::cout << "Ошибка, введите другое имя файла: ";
		std::string fname;
		std::cin >> fname;
		while (!check_file(fname))
		{
			std::cout << "Ошибка, введите другое имя файла: ";
			std::cin >> fname;
		}
		file.open(fname);
	}
}

void read_file_str(std::ifstream& file, std::string& str)
{
	file >> str;
	while (!condition_string(str))
	{
		file.close();
		std::cout << "Ошибка, введите другое имя файла: ";
		std::string fname;
		std::cin >> fname;
		while (!check_file(fname))
		{
			std::cout << "Ошибка, введите другое имя файла: ";
			std::cin >> fname;
		}
		file.open(fname);
		file >> str;
	}
}

void read_file_sec(std::ifstream& file, uShort& sec)
{
	while (!(file >> sec) || sec >= 86400)
	{
		file.close();
		std::cout << "Ошибка, введите другое имя файла: ";
		std::string fname;
		std::cin >> fname;
		while (!check_file(fname))
		{
			std::cout << "Ошибка, введите другое имя файла: ";
			std::cin >> fname;
		}	
		file.open(fname);
	}
}

void read_console_nums(uShort& hour, uShort& minute, uShort& second)
{
	read_and_check(hour, "Введите часы: ", 0, 23);
	read_and_check(minute, "Введите минуты: ", 0, 59);
	read_and_check(second, "Введите секунды: ", 0, 59);
}

void read_console_str(std::string& str)
{
	read_and_check(str, "Введите время в строковом представлении: ");
}

void read_console_sec(uShort& sec)
{
	read_and_check(sec, "Введите количество секунд: ", 0, 85399);
}

void action(Time& time)
{
	int choice = 0;
	do
	{
		choice = action_menu();
		switch (choice)
		{
		case 1: // вычесть секунды
		{
			uShort sec;
			switch (input_menu())
			{
			case 1:
				read_and_check(sec, "Введите количество секунд: ", 0, 85399);
				break;
			case 2:
			{
				std::string fname = choose_in_file_name();
				std::ifstream file(fname);
				read_file_sec(file, sec);
				console_output(time);
				break;
			}
			}
			time.subtract_seconds(sec);
			break;
		}
		case 2: // добавить секунды
		{
			uShort sec;
			switch (input_menu())
			{
			case 1:
				read_and_check(sec, "Введите количество секунд: ", 0, 85399);
				break;

			{
				std::string fname = choose_in_file_name();
				std::ifstream file(fname);
				read_file_sec(file, sec);
				console_output(time);
				break;
			}
			}
			time.add_seconds(sec);
			break;
		}
		case 3: // перевести время в секунды
			switch (output_menu())
			{
			case 1:
				std::cout << time.to_seconds() << '\n';
				break;
			case 2:
				std::string fname = choose_out_file_name();
				std::ofstream file(fname);
				file << time.to_seconds();
				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}
			break;

			switch (output_menu())
			{
			case 1:
				std::cout << time.to_minutes() << '\n';
				break;

				file << time.to_minutes();
				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}
			break;
		case 5: // вывести время на консоль
			console_output(time);
			break;
		case 6: // вывести время в файл
		{
			std::string fname = choose_out_file_name();

			break;
		}

			switch (output_menu())
			{
			case 1:
				std::cout << time.to_string() << '\n';
				break;
			case 2:
				std::string fname = choose_out_file_name();
				std::ofstream file(fname);
				file << time.to_seconds();
				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}
			break;
		case 8: // сложение двух моментов времени
		{
			Time new_time;
			initialization(new_time);
			Time res = time + new_time;
			std::cout << '\n';
			switch (output_menu())
			{
			case 1:
				console_output(res);
				std::cout << '\n';
				break;
			case 2:
				std::string fname = choose_out_file_name();
				std::ofstream file(fname);
				file_output(res, file);
				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}
			break;
		}
		case 9: // сложение секунд и времени
		{
			uShort sec;
			switch (input_menu())
			{
			case 1:
				read_and_check(sec, "Введите количество секунд: ", 0, 85399);
				break;
			case 2:
			{
				std::string fname = choose_in_file_name();
				std::ifstream file(fname);
				read_file_sec(file, sec);
				console_output(time);
				break;
			}
			}
			Time res = time + sec;
			std::cout << '\n';
			switch (output_menu())
			{
			case 1:
				console_output(res);
				std::cout << '\n';
				break;
			case 2:
				std::string fname = choose_out_file_name();
				std::ofstream file(fname);
				file_output(res, file);
				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}
			break;
		}
		case 10: // разность двух моментов времени
		{
			Time new_time;
			initialization(new_time);
			Time res = time - new_time;
			std::cout << '\n';
			switch (output_menu())
			{
			case 1:
				console_output(res);
				std::cout << '\n';
				break;
			case 2:
				std::string fname = choose_out_file_name();
				std::ofstream file(fname);
				file_output(res, file);
				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}
			break;
		}
		case 11: // разность времени и секунд
		{
			uShort sec;
			switch (input_menu())
			{
			case 1:
				read_and_check(sec, "Введите количество секунд: ", 0, 85399);
				break;
			case 2:
			{
				std::string fname = choose_in_file_name();

				read_file_sec(file, sec);

			switch (output_menu())

			case 1:
				console_output(res);
				std::cout << '\n';
				break;
			case 2:
				std::string fname = choose_out_file_name();
				std::ofstream file(fname);
				file_output(res, file);
				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}
			break;
		}
		case 12: // проверка на равенство
		{
			Time new_time;
			initialization(new_time);
			switch (output_menu())
			{
			case 1:
				std::cout << ((time == new_time) ? "true" : "false") << '\n';
				break;
			case 2:
				std::string fname = choose_out_file_name();

			}

		}
		case 13: // проверка на неравенство
		{
			Time new_time;
			initialization(new_time);
			switch (output_menu())
			{
			case 1:
				std::cout << ((time != new_time) ? "true" : "false") << '\n';

				std::string fname = choose_out_file_name();
				std::ofstream file(fname);

				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}
			break;
		}
		case 14: // первое больше второго
		{
			Time new_time;
			initialization(new_time);
			switch (output_menu())
			{
			case 1:
				std::cout << ((time > new_time) ? "true" : "false") << '\n';

				std::string fname = choose_out_file_name();

				file << ((time > new_time) ? "true" : "false");
				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}

		}
		case 15: // первое меньше второго
		{
			Time new_time;
			initialization(new_time);
			switch (output_menu())
			{
			case 1:
				std::cout << ((time < new_time) ? "true" : "false") << '\n';
				break;
			case 2:
				std::string fname = choose_out_file_name();
				std::ofstream file(fname);
				file << ((time < new_time) ? "true" : "false");
				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}
			break;
		}
		case 16: // первое больше или равно второму
		{
			Time new_time;
			initialization(new_time);
			switch (output_menu())
			{
			case 1:
				std::cout << ((time >= new_time) ? "true" : "false") << '\n';
				break;
			case 2:
				std::string fname = choose_out_file_name();

				std::cout << "Информация напечатана в файл " << fname << '\n';
				file.close();
				break;
			}

		}
		case 17: // первое меньше или равно второму
		{
			Time new_time;
			initialization(new_time);
			switch (output_menu())
			{
			case 1:

				break;
			case 2:
				std::string fname = choose_out_file_name();
				std::ofstream file(fname);
				file << ((time <= new_time) ? "true" : "false");

}

}
